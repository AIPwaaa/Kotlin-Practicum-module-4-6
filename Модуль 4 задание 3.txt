data class GithubRepo(
    val id: Int,
    val full_name: String,
    val description: String?,
    val stargazers_count: Int,
    val language: String?
)

fun loadRepos(context: Context): List<GithubRepo> {
    val json = context.assets
        .open("github_repos.json")
        .bufferedReader()
        .use { it.readText() }

    val type = object : com.google.gson.reflect.TypeToken<List<GithubRepo>>() {}.type
    return com.google.gson.Gson().fromJson(json, type)
}

@Composable
fun GithubSearchScreen() {
    val context = LocalContext.current
    val scope = rememberCoroutineScope()

    var query by remember { mutableStateOf("") }
    var results by remember { mutableStateOf<List<GithubRepo>>(emptyList()) }
    var isLoading by remember { mutableStateOf(false) }

    var searchJob by remember { mutableStateOf<Job?>(null) }

    var allRepos = remember { loadRepos(context) }

    Column(modifier = Modifier.fillMaxSize().padding(16.dp)) {
        OutlinedTextField(
            value = query,
            onValueChange = { neText ->
                query = neText

                searchJob?.cancel()

                searchJob = scope.launch {
                    delay(500)

                    isLoading = true

                    val filtered = withContext(Dispatchers.Default) {
                        allRepos.filter {
                            it.full_name.contains(query, ignoreCase = true)
                        }
                    }
                    results = filtered
                    isLoading = false
                }
            },
            label = { Text("Search repository") },
            modifier = Modifier.fillMaxWidth()
        )
        Spacer(Modifier.height(16.dp))

        if (isLoading) {
            CircularProgressIndicator()
        }

        LazyColumn {
            items(results) { repo ->
                RepoItem(repo)
            }
        }
    }
}

@Composable
fun RepoItem(repo: GithubRepo) {
    Column(modifier = Modifier.fillMaxWidth().padding(8.dp)) {
        Text(text = repo.full_name, fontWeight = FontWeight.Bold)
        Text(text = repo.description ?: "")
        Text(text = "‚≠ê ${repo.stargazers_count}")
        Divider()
    }
}
