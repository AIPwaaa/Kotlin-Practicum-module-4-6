class MainActivity : ComponentActivity() {

    private lateinit var fusedLocationClient: FusedLocationProviderClient

    private val addressText = mutableStateOf("Адрес не получен")
    private val coordinatesText = mutableStateOf("")
    private val isLoading = mutableStateOf(false)

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        fusedLocationClient = LocationServices.getFusedLocationProviderClient(this)

        setContent {
            Screen(
                address = addressText.value,
                coordinates = coordinatesText.value,
                isLoading = isLoading.value,
                onGetAddressClick = { getAddress() }
            )
        }
    }

    private fun getAddress() {
        isLoading.value = true

        val fine = checkSelfPermission(Manifest.permission.ACCESS_FINE_LOCATION) == PackageManager.PERMISSION_GRANTED
        val coarse = checkSelfPermission(Manifest.permission.ACCESS_COARSE_LOCATION) == PackageManager.PERMISSION_GRANTED

        if (!fine && !coarse) {
            val permissionLauncher =
                registerForActivityResult(ActivityResultContracts.RequestMultiplePermissions()) { permissions ->
                    val grantedFine = permissions[Manifest.permission.ACCESS_FINE_LOCATION] == true
                    val grantedCoarse = permissions[Manifest.permission.ACCESS_COARSE_LOCATION] == true

                    if (grantedFine || grantedCoarse) {
                        getAddress()
                    } else {
                        isLoading.value = false
                        addressText.value = "Разрешение на геолокацию отклонено"
                    }
                }

            permissionLauncher.launch(
                arrayOf(
                    Manifest.permission.ACCESS_FINE_LOCATION,
                    Manifest.permission.ACCESS_COARSE_LOCATION
                )
            )
            return
        }

        fusedLocationClient.getCurrentLocation(
            Priority.PRIORITY_HIGH_ACCURACY,
            null
        ).addOnSuccessListener { location ->
            if (location != null) {
                val lat = location.latitude
                val lng = location.longitude
                coordinatesText.value = "lat: $lat  lng: $lng"

                // Geocoder
                val geocoder = Geocoder(this, Locale.getDefault())

                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                    geocoder.getFromLocation(lat, lng, 1) { addresses ->
                        if (addresses.isNotEmpty()) {
                            val fullAddress = addresses[0].getAddressLine(0)
                            addressText.value = fullAddress
                        } else {
                            addressText.value = "Адрес не найден"
                        }
                        isLoading.value = false
                    }
                } else {
                    val addresses = geocoder.getFromLocation(lat, lng, 1)
                    if (addresses != null && addresses.isNotEmpty()) {
                        val fullAddress = addresses[0].getAddressLine(0)
                        addressText.value = fullAddress
                    } else {
                        addressText.value = "Адрес не найден"
                    }
                    isLoading.value = false
                }
            } else {
                addressText.value = "Не удалось получить координаты"
                isLoading.value = false
            }
        }.addOnFailureListener {
            addressText.value = "Ошибка при получении координат"
            isLoading.value = false
        }
    }
}

@Composable
fun Screen(
    address: String,
    coordinates: String,
    isLoading: Boolean,
    onGetAddressClick: () -> Unit
) {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(20.dp),
        verticalArrangement = Arrangement.Center,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Button(onClick = onGetAddressClick) {
            Text(text = "Получить мой адрес")
        }

        if (isLoading) {
            CircularProgressIndicator()
            Spacer(modifier = Modifier.height(20.dp))
        }

        Text(
            text = address,
            fontSize = 22.sp
        )

        Text(text = coordinates)
    }
}
