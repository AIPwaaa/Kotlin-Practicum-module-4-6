@Composable
fun TimerScreen() {

    val context = LocalContext.current
    val workManager = WorkManager.getInstance(context)
    var status by remember { mutableStateOf("Ожидание...") }
    var compressId by remember { mutableStateOf<UUID?>(null) }
    var watermarkId by remember { mutableStateOf<UUID?>(null) }
    var uploadId by remember { mutableStateOf<UUID?>(null) }
    val compressInfo by compressId?.let {workManager.getWorkInfoByIdLiveData(it).observeAsState()} ?: remember {mutableStateOf(null)}
    val watermarkInfo by watermarkId?.let {workManager.getWorkInfoByIdLiveData(it).observeAsState()} ?: remember {mutableStateOf(null)}
    val uploadInfo by uploadId?.let {
        workManager.getWorkInfoByIdLiveData(it).observeAsState()
    } ?: remember {mutableStateOf(null)}


    compressInfo?.progress?.getString("status")?.let {status = it}
    watermarkInfo?.progress?.getString("status")?.let {status = it}
    uploadInfo?.let {
        status = it.progress.getString("status") ?: status
        if (it.state == WorkInfo.State.SUCCEEDED) {
            status = it.outputData.getString("status") ?: status
        }
    }

    Column(
        modifier = Modifier.fillMaxSize(),
        horizontalAlignment = Alignment.CenterHorizontally,
        verticalArrangement = Arrangement.Center
    ) {
        Text(text=status)

        if (compressInfo?.state == WorkInfo.State.RUNNING
            || watermarkInfo?.state == WorkInfo.State.RUNNING
            || uploadInfo?.state == WorkInfo.State.RUNNING
        ) {
            LinearProgressIndicator(modifier = Modifier.fillMaxWidth())
        }

        Button(onClick = {
            val compressWork = OneTimeWorkRequestBuilder<CompressWorker>().build()
            val watermarkWork = OneTimeWorkRequestBuilder<WatermarkWorker>().build()
            val uploadWork = OneTimeWorkRequestBuilder<UploadWorker>().build()

            compressId = compressWork.id
            watermarkId = watermarkWork.id
            uploadId = uploadWork.id

            workManager
                .beginWith(compressWork)
                .then(watermarkWork)
                .then(uploadWork)
                .enqueue()

            if(compressInfo?.state == WorkInfo.State.FAILED
                || watermarkInfo?.state == WorkInfo.State.FAILED
                || uploadInfo?.state == WorkInfo.State.FAILED) {
                status = "Ошибка"
            }
        },
            enabled = compressInfo?.state != WorkInfo.State.RUNNING
                    && watermarkInfo?.state != WorkInfo.State.RUNNING
                    && uploadInfo?.state != WorkInfo.State.RUNNING
        ) {
            Text("Загрузить")
        }
    }
}
