data class Post(
    val id: Int,
    val userId: Int,
    val title: String,
    val body: String,
    val avatarUrl: String
)

data class Comment(
    val postId: Int,
    val id: Int,
    val name: String,
    val body: String
)

sealed class PostUiState {
    object Loading : PostUiState()
    data class Ready(
        val post: Post,
        val avatarColor: Color,
        val comments: List<Comment>
    ) : PostUiState()
    object Error : PostUiState()
}

@Composable
fun SocialFeedScreen() {

    val scope = rememberCoroutineScope()
    val context = LocalContext.current

    var postsState by remember { mutableStateOf<List<PostUiState>>(emptyList()) }
    var parentJob by remember { mutableStateOf<Job?>(null) }

    fun load() {
        parentJob?.cancel()

        parentJob = scope.launch {

            val posts = withContext(Dispatchers.IO) {
                loadPosts(context)
            }

            postsState = posts.map { PostUiState.Loading }

            posts.forEachIndexed { index, post ->

                launch {

                    postsState = postsState.toMutableList().also {
                        it[index] = PostUiState.Loading
                    }

                    try {
                        supervisorScope {

                            val avatarDeferred = async {
                                delay(800)
                                Color(
                                    Random.nextFloat(),
                                    Random.nextFloat(),
                                    Random.nextFloat()
                                )
                            }

                            val commentsDeferred = async {
                                delay(1200)
                                loadComments(context)
                                    .filter { it.postId == post.id }
                            }

                            val avatar = avatarDeferred.await()
                            val comments = commentsDeferred.await()

                            postsState = postsState.toMutableList().also {
                                it[index] =
                                    PostUiState.Ready(post, avatar, comments)
                            }
                        }
                    } catch (e: Exception) {
                        postsState = postsState.toMutableList().also {
                            it[index] = PostUiState.Error
                        }
                    }
                }
            }
        }
    }

    LaunchedEffect(Unit) {
        load()
    }

    Column {

        Button(onClick = { load() }) {
            Text("Refresh")
        }

        LazyColumn {
            items(postsState) { state ->
                PostCard(state)
            }
        }
    }
}

@Composable
fun PostCard(state: PostUiState) {

    Card(
        modifier = Modifier
            .fillMaxWidth()
            .padding(8.dp)
    ) {
        when (state) {

            is PostUiState.Loading -> {
                Box(
                    Modifier
                        .fillMaxWidth()
                        .padding(16.dp),
                    contentAlignment = Alignment.Center
                ) {
                    CircularProgressIndicator()
                }
            }

            is PostUiState.Error -> {
                Text(
                    "Error loading post",
                    modifier = Modifier.padding(16.dp),
                    color = Color.Red
                )
            }

            is PostUiState.Ready -> {
                Column(Modifier.padding(16.dp)) {

                    Box(
                        Modifier
                            .size(50.dp)
                            .background(state.avatarColor, CircleShape)
                    )

                    Spacer(Modifier.height(8.dp))

                    Text(state.post.title, fontWeight = FontWeight.Bold)
                    Text(state.post.body)

                    Spacer(Modifier.height(8.dp))

                    state.comments.forEach {
                        Text("- ${it.name}: ${it.body}")
                    }
                }
            }
        }
    }
}

fun loadPosts(context: Context): List<Post> {
    val json = context.assets.open("social_posts.json")
        .bufferedReader()
        .use { it.readText() }

    val type = object : TypeToken<List<Post>>() {}.type
    return Gson().fromJson(json, type)
}

fun loadComments(context: Context): List<Comment> {
    val json = context.assets.open("comments.json")
        .bufferedReader()
        .use { it.readText() }

    val type = object : TypeToken<List<Comment>>() {}.type
    return Gson().fromJson(json, type)
}
